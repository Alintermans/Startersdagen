Opdracht 4.6

Pas je code nu aan zodat de servomotor start bij een hoek van 90°. Vervolgens moet hij langzaam draaien naar 120° om daarna terug naar 90° te draaien. Hieronder kan je een beetje inspiratie opdoen. 

              
void loop() {
  if (Serial.available() > 0) {
    char letter = Serial.read();
    if (letter == 's') {
      Serial.println("Servo motor begint nu");
      for (i=90; i<=120; i++) {
        myservo.write(i);
        delay(20);
      }
      delay(200);
      for (i=120; i>=90; i--) {
        myservo.write(i);
        delay(20);
      }
    }
  }
}
