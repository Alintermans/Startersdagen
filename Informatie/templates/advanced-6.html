Opdracht 4.3

Pas je code nu aan zodat de servomotor start bij een hoek van 90°. Vervolgens draait hij langzaam draaien naar 120° om daarna terug naar 90° te draaien. Hieronder kan je inspiratie opdoen. 

              
int start = 1;
void loop() {
  if (start == 1) {
    for (i=90; i<=120; i++) {
      myservo.write(i);
      delay(20);
    }
    delay(200);
    for (i=120; i>=90; i--) {
      myservo.write(i);
      delay(20);
    }
    start = 0;
  }
}
